    export default {
  data() {
    return {
      selected: 'sport'
    }
  },
  watch: {
    selected(newValue) {
      if (newValue === 'sport') {
        alert(newValue)
      }
    }
  },
  methods: {
    clickFn(name) {
      if (name === 'music') {
        alert(name)
      }
    }
  }
}

    # API

    # nav

    参数 说明 类型 默认值 可选值
    selected 选中的菜单项的 name 值，必需字段 String - -
    vertical 菜单类型，是否支持垂直 Boolean false true
    @update:selected="clickFn" 点击菜单时的回调 Function - -

    # subNav

    参数 说明 类型 默认值 可选值
    name 嵌套子菜单名字，必需字段 String - -

    # navItem

    参数 说明 类型 默认值 可选值
    name 子菜单名字，必需字段 String - -

    TIP

    点击菜单时，会触发 selected 字段值的更新，可在此时做相应的逻辑处理
